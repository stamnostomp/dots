[GTK 3]

COPY THEMES TO:
User wide theme: ~/.themes
System wide theme: ~/.local/share/themes/

COPY ICONS TO:
User wide icons: ~/.icons
System wide theme: ~/.local/share/icons/

COPY CFG TO:
System CFG: /etc/gtk-3.0/settings.ini
User CFG: $XDG_CONFIG_HOME/gtk-3.0/settings.ini


[GTK 2]

COPY THEMES TO:
same as gtk 3

COPY ICONS TO:
same as gtk 3

COPY CFG TO:
System CFG: /etc/gtk-2.0/gtkrc
User CFG: ~/.gtkrc-2.0/gtkrc
